    Script function Select1pMode(self)
	self.Is1pModeSelected = true
	self.Is3pModeSelected = false
	self.IsControllable3pModeSelected = false
	self.WasForcedInto3pFrom1p = false
	self.FireEvent("OnSelect1pMode")
end

function ForceSelect3pMode(self)
	self.WasForcedInto3pFrom1p = true
	self.Is3pModeSelected = true
	self.Is1pModeSelected = false
	self.IsControllable3pModeSelected = false
	self.FireEvent("OnSelect3pMode")
end

--function <guid>( self, event, vars, sharedvars, globalvars, deltaTime )
do
	if event == "Init" then
		-- default mode for player view is 1p
		self.Is1pModeSelected = true
		self.Is3pModeSelected = false
		self.IsControllable3pModeSelected = false
		self.WasForcedInto3pFrom1p = false
		return
	elseif event == "Select1pMode" then
		if self.PlayerIsInVehicle and not self.VehicleSeatHas1pCam then
			-- force select 3p mode since there is no 1p camera in the occupied vehicle seat
			ForceSelect3pMode(self)
		else 
			-- normal 1p mode selection
			Select1pMode(self)
		end
		return
	elseif event == "Select3pMode" then
		self.Is3pModeSelected = true
		self.Is1pModeSelected = false
		self.IsControllable3pModeSelected = false
		self.WasForcedInto3pFrom1p = false
		self.FireEvent("OnSelect3pMode")
		return
	elseif event == "SelectControllable3pMode" then
		self.IsControllable3pModeSelected = true
		self.Is1pModeSelected = false
		self.Is3pModeSelected = false
		self.WasForcedInto3pFrom1p = false
		self.FireEvent("OnSelectControllable3pMode")
		return
	elseif event == "OnEnterWaitingForViewState" then
		if (self.WasSpecForcedInto3pFrom1p and ((not self.PlayerIsInVehicle and self.PlayerHasValidSoldier) or (self.PlayerIsInVehicle and self.VehicleSeatHas1pCam))) then
				-- force select 1p mode since 3p was previously forced
				Select1pMode(self)
				return
		end
	elseif event == "OnPlayerViewUpdated" then
		if (self.IsProjCamActive and self.Is1pModeSelected) or (self.PlayerIsInVehicle and self.Is1pModeSelected and not self.VehicleSeatHas1pCam) then
			-- force select 3p mode since there is no 1p camera in the occupied vehicle seat
			ForceSelect3pMode(self)
			return
		elseif not self.IsProjCamActive and self.WasSpecForcedInto3pFrom1p and self.Is3pModeSelected and (self.IsSpectatingSoldier or (self.IsSpectatingVehicle and self.VehicleSeatHas1pCam)) then 
			-- force select 1p mode since 3p was previously forced
			Select1pMode(self)
			return
		end
	elseif event == "ForceSelect3pMode" then
		ForceSelect3pMode(self)
		return		
	end
end
